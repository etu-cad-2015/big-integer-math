// Остаток от деления большего натурального числа на меньшее или равное натуральное с остатком (делитель отличен от нуля)
// Выполнил: Клюкин Александр, 5302
// Модуль: N-12
function MOD_NN_N (A, B) { // на вход два числа
	var C; // в этой переменной хранится результат - остаток от деления.
	if (A >= B) // проверка на большее число
		C = A%B;
	else
		C = B%A;
	return C; // возврат остатка от деления
}
